💻 에러 방지 코드 예시

SANS Kids만의 특별한 에러 방지 패턴으로 안전하게 코딩해요 

// 🛡️ 항상 try-catch로 게임 로직을 보호해요
function safeGameLoop() {
    try {
        updateGame();    // 게임 업데이트
        drawGame();      // 화면 그리기
    } catch (error) {
        // 에러가 나도 게임은 계속!
        showMessage("🔄 다시 시도 중...");
    }
    requestAnimationFrame(safeGameLoop);
}

// 🎯 기본값으로 안전하게
const hero = hero || { 
    x: 100, 
    y: 300, 
    size: 50,
    color: '#3498db'
};

// 💬 시각적 피드백으로 디버깅 대체
function showMessage(text) {
    document.getElementById('status').innerText = text;
}
